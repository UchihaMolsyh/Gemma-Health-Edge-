import 'dart:async';
import 'dart:convert';
import 'package:http/http.dart' as http;

class ApiClient {
  String serverUrl;
  final Map<String, String> _headers = {'Content-Type': 'application/json'};
  
  // Connection metrics
  int _totalRequests = 0;
  int _successfulRequests = 0;
  int _totalLatencyMs = 0;
  final List<int> _recentLatencies = [];
  static const int _maxRecentLatencies = 10;

  ApiClient({required this.serverUrl});

  Future<bool> checkHealth() async {
    try {
      final start = DateTime.now();
      final response = await http.get(
        Uri.parse('$serverUrl/health'),
        headers: _headers,
      ).timeout(const Duration(seconds: 5));
      final latency = DateTime.now().difference(start).inMilliseconds;
      
      _updateMetrics(response.statusCode == 200, latency);
      return response.statusCode == 200;
    } catch (e) {
      _updateMetrics(false, 0);
      return false;
    }
  }

  Future<String?> autoDetectServer() async {
    final localIp = '192.168.1';
    for (int i = 1; i <= 255; i++) {
      final url = 'http://$localIp.$i:8080';
      try {
        final response = await http.get(
          Uri.parse('$url/health'),
          headers: _headers,
        ).timeout(const Duration(milliseconds: 200));
        if (response.statusCode == 200) {
          serverUrl = url;
          return url;
        }
      } catch (_) {
        continue;
      }
    }
    return null;
  }

  Stream<String> streamChat({
    required List<Map<String, dynamic>> messages,
    required bool useCloudApi,
    required String cloudApiKey,
    required String cloudModelId,
  }) async* {
    if (useCloudApi) {
      yield* _streamCloudChat(messages, cloudApiKey, cloudModelId);
    } else {
      yield* _streamLocalChat(messages);
    }
  }

  Stream<String> _streamLocalChat(List<Map<String, dynamic>> messages) async* {
    try {
      final start = DateTime.now();
      final request = http.Request('POST', Uri.parse('$serverUrl/v1/chat/completions'));
      request.headers.addAll(_headers);
      request.body = jsonEncode({
        'model': 'gemma-2-27b-it',
        'messages': messages,
        'stream': true,
        'max_tokens': 2048,
      });

      final response = await request.send().timeout(const Duration(seconds: 60));
      final latency = DateTime.now().difference(start).inMilliseconds;
      _updateMetrics(response.statusCode == 200, latency);

      if (response.statusCode != 200) {
        yield '[ERROR] Server returned ${response.statusCode}';
        return;
      }

      final stream = response.stream.transform(utf8.decoder);
      await for (final chunk in stream) {
        final lines = chunk.split('\n');
        for (final line in lines) {
          if (line.startsWith('data: ')) {
            final data = line.substring(6);
            if (data == '[DONE]') continue;
            try {
              final json = jsonDecode(data);
              final content = json['choices']?[0]?['delta']?['content'];
              if (content != null) {
                yield content;
              }
            } catch (_) {}
          }
        }
      }
    } catch (e) {
      _updateMetrics(false, 0);
      yield '[ERROR] Connection failed: $e';
    }
  }

  Stream<String> _streamCloudChat(
    List<Map<String, dynamic>> messages,
    String apiKey,
    String modelId,
  ) async* {
    try {
      final start = DateTime.now();
      final request = http.Request('POST', Uri.parse('https://openrouter.ai/api/v1/chat/completions'));
      request.headers.addAll({
        'Content-Type': 'application/json',
        'Authorization': 'Bearer $apiKey',
      });
      request.body = jsonEncode({
        'model': modelId,
        'messages': messages,
        'stream': true,
        'max_tokens': 2048,
      });

      final response = await request.send().timeout(const Duration(seconds: 60));
      final latency = DateTime.now().difference(start).inMilliseconds;
      _updateMetrics(response.statusCode == 200, latency);

      if (response.statusCode != 200) {
        yield '[ERROR] Cloud API returned ${response.statusCode}';
        return;
      }

      final stream = response.stream.transform(utf8.decoder);
      await for (final chunk in stream) {
        final lines = chunk.split('\n');
        for (final line in lines) {
          if (line.startsWith('data: ')) {
            final data = line.substring(6);
            if (data == '[DONE]') continue;
            try {
              final json = jsonDecode(data);
              final content = json['choices']?[0]?['delta']?['content'];
              if (content != null) {
                yield content;
              }
            } catch (_) {}
          }
        }
      }
    } catch (e) {
      _updateMetrics(false, 0);
      yield '[ERROR] Cloud connection failed: $e';
    }
  }

  Future<String> completeChat({
    required List<Map<String, dynamic>> messages,
    required bool useCloudApi,
    required String cloudApiKey,
    required String cloudModelId,
    int maxTokens = 64,
    Duration timeout = const Duration(seconds: 30),
  }) async {
    if (useCloudApi) {
      return await _completeCloudChat(messages, cloudApiKey, cloudModelId, maxTokens, timeout);
    } else {
      return await _completeLocalChat(messages, maxTokens, timeout);
    }
  }

  Future<String> _completeLocalChat(
    List<Map<String, dynamic>> messages,
    int maxTokens,
    Duration timeout,
  ) async {
    try {
      final start = DateTime.now();
      final response = await http.post(
        Uri.parse('$serverUrl/v1/chat/completions'),
        headers: _headers,
        body: jsonEncode({
          'model': 'gemma-2-27b-it',
          'messages': messages,
          'stream': false,
          'max_tokens': maxTokens,
        }),
      ).timeout(timeout);
      final latency = DateTime.now().difference(start).inMilliseconds;
      _updateMetrics(response.statusCode == 200, latency);

      if (response.statusCode != 200) {
        return 'ERROR: ${response.statusCode}';
      }

      final json = jsonDecode(response.body);
      return json['choices']?[0]?['message']?['content'] ?? '';
    } catch (e) {
      _updateMetrics(false, 0);
      return 'ERROR: $e';
    }
  }

  Future<String> _completeCloudChat(
    List<Map<String, dynamic>> messages,
    String apiKey,
    String modelId,
    int maxTokens,
    Duration timeout,
  ) async {
    try {
      final start = DateTime.now();
      final response = await http.post(
        Uri.parse('https://openrouter.ai/api/v1/chat/completions'),
        headers: {
          'Content-Type': 'application/json',
          'Authorization': 'Bearer $apiKey',
        },
        body: jsonEncode({
          'model': modelId,
          'messages': messages,
          'stream': false,
          'max_tokens': maxTokens,
        }),
      ).timeout(timeout);
      final latency = DateTime.now().difference(start).inMilliseconds;
      _updateMetrics(response.statusCode == 200, latency);

      if (response.statusCode != 200) {
        return 'ERROR: ${response.statusCode}';
      }

      final json = jsonDecode(response.body);
      return json['choices']?[0]?['message']?['content'] ?? '';
    } catch (e) {
      _updateMetrics(false, 0);
      return 'ERROR: $e';
    }
  }

  Future<String?> fetchWikipediaSummary(String searchTerm) async {
    try {
      final url = 'https://en.wikipedia.org/api/rest_v1/page/summary/$searchTerm';
      final response = await http.get(Uri.parse(url)).timeout(const Duration(seconds: 10));
      if (response.statusCode == 200) {
        final json = jsonDecode(response.body);
        return json['extract'];
      }
    } catch (_) {}
    return null;
  }

  void _updateMetrics(bool success, int latencyMs) {
    _totalRequests++;
    if (success) {
      _successfulRequests++;
      _totalLatencyMs += latencyMs;
      _recentLatencies.add(latencyMs);
      if (_recentLatencies.length > _maxRecentLatencies) {
        _recentLatencies.removeAt(0);
      }
    }
  }

  Map<String, dynamic> getConnectionMetrics() {
    final successRate = _totalRequests > 0
        ? '${(_successfulRequests / _totalRequests * 100).toStringAsFixed(0)}%'
        : 'N/A';
    final avgLatency = _successfulRequests > 0
        ? (_totalLatencyMs / _successfulRequests).round()
        : 0;
    return {
      'successRate': successRate,
      'averageLatencyMs': avgLatency,
      'totalRequests': _totalRequests,
    };
  }
}

import 'dart:async';
import 'package:flutter_blue_plus/flutter_blue_plus.dart';
import 'models/health_sample.dart';

class BleService {
  static const heartRateServiceUuid = '0000180d-0000-1000-8000-00805f9b34fb';
  static const heartRateMeasurementUuid = '00002a37-0000-1000-8000-00805f9b34fb';
  static const bloodPressureServiceUuid = '00001810-0000-1000-8000-00805f9b34fb';
  static const bloodPressureMeasurementUuid = '00002a35-0000-1000-8000-00805f9b34fb';
  static const pulseOxServiceUuid = '00001822-0000-1000-8000-00805f9b34fb';
  static const pulseOxMeasurementUuid = '00002a5e-0000-1000-8000-00805f9b34fb';

  Stream<List<ScanResult>> scan({Duration timeout = const Duration(seconds: 10)}) async* {
    await FlutterBluePlus.startScan(timeout: timeout);
    yield* FlutterBluePlus.scanResults;
  }

  Future<void> stopScan() async {
    await FlutterBluePlus.stopScan();
  }

  Future<BluetoothDevice> connect(BluetoothDevice device) async {
    await device.connect();
    await device.discoverServices();
    return device;
  }

  Future<void> disconnect(BluetoothDevice device) async {
    await device.disconnect();
  }

  Stream<HealthSample> subscribeMeasurements(BluetoothDevice device) async* {
    final services = device.services;
    if (services == null) return;

    for (final service in services) {
      if (service.uuid.toString().toLowerCase() == heartRateServiceUuid) {
        yield* _subscribeHeartRate(service);
      } else if (service.uuid.toString().toLowerCase() == bloodPressureServiceUuid) {
        yield* _subscribeBloodPressure(service);
      } else if (service.uuid.toString().toLowerCase() == pulseOxServiceUuid) {
        yield* _subscribePulseOx(service);
      }
    }
  }

  Stream<HealthSample> _subscribeHeartRate(BluetoothService service) async* {
    final characteristic = service.characteristics.firstWhere(
      (c) => c.uuid.toString().toLowerCase() == heartRateMeasurementUuid,
      orElse: () => throw Exception('Heart rate characteristic not found'),
    );
    await characteristic.setNotifyValue(true);
    yield* characteristic.value.map((data) {
      return _parseHeartRate(data);
    });
  }

  Stream<HealthSample> _subscribeBloodPressure(BluetoothService service) async* {
    final characteristic = service.characteristics.firstWhere(
      (c) => c.uuid.toString().toLowerCase() == bloodPressureMeasurementUuid,
      orElse: () => throw Exception('Blood pressure characteristic not found'),
    );
    await characteristic.setNotifyValue(true);
    yield* characteristic.value.map((data) {
      return _parseBloodPressure(data);
    });
  }

  Stream<HealthSample> _subscribePulseOx(BluetoothService service) async* {
    final characteristic = service.characteristics.firstWhere(
      (c) => c.uuid.toString().toLowerCase() == pulseOxMeasurementUuid,
      orElse: () => throw Exception('Pulse ox characteristic not found'),
    );
    await characteristic.setNotifyValue(true);
    yield* characteristic.value.map((data) {
      return _parsePulseOx(data);
    });
  }

  HealthSample _parseHeartRate(List<int> data) {
    if (data.isEmpty) return HealthSample(type: 'HEART_RATE', value: 0, unit: 'bpm');
    
    final flags = data[0];
    final is16Bit = (flags & 0x01) == 0;
    
    int heartRate;
    if (is16Bit && data.length >= 3) {
      heartRate = (data[1] + (data[2] << 8));
    } else if (data.length >= 2) {
      heartRate = data[1];
    } else {
      heartRate = 0;
    }
    
    return HealthSample(
      type: 'HEART_RATE',
      value: heartRate.toDouble(),
      unit: 'bpm',
      source: 'BLE',
    );
  }

  HealthSample _parseBloodPressure(List<int> data) {
    if (data.length < 7) {
      return HealthSample(type: 'BLOOD_PRESSURE_SYSTOLIC', value: 0, unit: 'mmHg');
    }
    
    final systolic = data[1] + (data[2] << 8);
    final diastolic = data[3] + (data[4] << 8);
    
    return HealthSample(
      type: 'BLOOD_PRESSURE_SYSTOLIC',
      value: systolic.toDouble(),
      unit: 'mmHg',
      source: 'BLE',
    );
  }

  HealthSample _parsePulseOx(List<int> data) {
    if (data.length < 5) {
      return HealthSample(type: 'BLOOD_OXYGEN', value: 0, unit: '%');
    }
    
    final spo2 = data[2];
    final pulse = data[1] + (data[0] << 8);
    
    return HealthSample(
      type: 'BLOOD_OXYGEN',
      value: spo2.toDouble(),
      unit: '%',
      source: 'BLE',
    );
  }
}

class HealthService {
  final Health _health = Health();

  Future<bool> requestAuthorization() async {
    final types = [
      HealthDataType.STEPS,
      HealthDataType.HEART_RATE,
      HealthDataType.BLOOD_OXYGEN,
      HealthDataType.BLOOD_PRESSURE_SYSTOLIC,
      HealthDataType.BLOOD_PRESSURE_DIASTOLIC,
    ];

    try {
      final requested = await _health.requestAuthorization(types);
      return requested;
    } catch (e) {
      return false;
    }
  }

  Future<List<HealthSample>> fetchSamples({
    required DateTime start,
    required DateTime end,
  }) async {
    final samples = <HealthSample>[];

    try {
      final steps = await _health.getHealthDataFromTypes(start, end, [HealthDataType.STEPS]);
      for (final s in steps) {
        if (s.value is num) {
          samples.add(HealthSample(
            type: 'STEPS',
            value: (s.value as num).toDouble(),
            unit: 'count',
            timestamp: s.dateFrom,
            source: s.sourceId ?? 'Health',
          ));
        }
      }

      final hr = await _health.getHealthDataFromTypes(start, end, [HealthDataType.HEART_RATE]);
      for (final s in hr) {
        if (s.value is num) {
          samples.add(HealthSample(
            type: 'HEART_RATE',
            value: (s.value as num).toDouble(),
            unit: 'bpm',
            timestamp: s.dateFrom,
            source: s.sourceId ?? 'Health',
          ));
        }
      }

      final spo2 = await _health.getHealthDataFromTypes(start, end, [HealthDataType.BLOOD_OXYGEN]);
      for (final s in spo2) {

          samples.add(HealthSample(

            unit: 'mmHg',
            timestamp: s.dateFrom,

import 'dart:async';
import 'dart:convert';
import 'package:flutter/services.dart';

class RagService {
  bool isInitialized = false;
  final Map<String, List<String>> _invertedIndex = {};
  final List<String> _documents = [];

  Future<void> initialize() async {
    try {
      final csvData = await rootBundle.loadString('assets/medical_kb.csv');
      final lines = csvData.split('\n');
      
      for (int i = 1; i < lines.length; i++) {
        final line = lines[i].trim();
        if (line.isEmpty) continue;
        
        final parts = line.split(',');
        if (parts.length >= 2) {
          final text = parts.sublist(1).join(',').trim();
          _documents.add(text);
          
          final words = text.toLowerCase().split(RegExp(r'\s+'));
          for (final word in words) {
            if (word.length > 2) {
              _invertedIndex.putIfAbsent(word, () => []);
              if (!_invertedIndex[word]!.contains(text)) {
                _invertedIndex[word]!.add(text);
              }
            }
          }
        }
      }
      
      isInitialized = true;
    } catch (e) {
      // Fallback: use empty index
      isInitialized = true;
    }
  }

  List<RagResult> search(String query, {int limit = 5}) {
    if (!isInitialized) return [];
    
    final queryWords = query.toLowerCase().split(RegExp(r'\s+'));
    final scores = <String, double>{};
    
    for (final word in queryWords) {
      if (word.length <= 2) continue;
      
      final matches = _invertedIndex[word] ?? [];
      for (final doc in matches) {
        scores[doc] = (scores[doc] ?? 0) + 1.0;
      }
    }
    
    final sorted = scores.entries.toList()
      ..sort((a, b) => b.value.compareTo(a.value));
    
    return sorted.take(limit).map((e) => RagResult(text: e.key, score: e.value)).toList();
  }
}

class RagResult {
  final String text;
  final double score;
  
  RagResult({required this.text, required this.score});
}

import 'package:flutter_secure_storage/flutter_secure_storage.dart';

class SecureStorageService {
  static const _storage = FlutterSecureStorage();
  static const _xorKey = 42; // Simple XOR for demo (not production-secure)

  Future<void> storeApiKey(String key, String value) async {
    final encrypted = _xorEncrypt(value);
    await _storage.write(key: key, value: encrypted);
  }

  Future<String?> getApiKey(String key) async {
    final encrypted = await _storage.read(key: key);
    if (encrypted == null) return null;
    return _xorDecrypt(encrypted);
  }

  Future<void> deleteApiKey(String key) async {
    await _storage.delete(key: key);
  }

  String _xorEncrypt(String text) {
    final bytes = text.codeUnits;
    final encrypted = bytes.map((b) => b ^ _xorKey).toList();
    return String.fromCharCodes(encrypted);
  }

  String _xorDecrypt(String encrypted) {
    final bytes = encrypted.codeUnits;
    final decrypted = bytes.map((b) => b ^ _xorKey).toList();
    return String.fromCharCodes(decrypted);
  }
}

import 'package:hive_flutter/hive_flutter.dart';
import 'models/app_settings.dart';
import 'models/health_sample.dart';
import 'models/message.dart';
import 'models/mood_entry.dart';
import 'models/session.dart';
import 'secure_storage_service.dart';

class StorageService {
  static const String _settingsKey = 'app_settings';
  static const String _sessionsKey = 'chat_sessions';
  static const String _moodsKey = 'mood_entries';
  static const String _healthKey = 'health_samples';
  
  final _secureStorage = SecureStorageService();
  late final Box _box;

  StorageService() {
    _box = Hive.box('gemma_health');
  }

  // Settings
  Future<AppSettings> loadSettings() async {
    final json = _box.get(_settingsKey);
    if (json == null) return const AppSettings();
    
    try {
      final settingsJson = json as Map<String, dynamic>;
      final apiKey = await _secureStorage.getApiKey('cloud_api_key');
      if (apiKey != null) {
        settingsJson['cloudApiKey'] = apiKey;
      }
      return AppSettings.fromJson(settingsJson);
    } catch (e) {
      return const AppSettings();
    }
  }

  Future<void> saveSettings(AppSettings settings) async {
    await _secureStorage.storeApiKey('cloud_api_key', settings.cloudApiKey);
    final json = settings.toJson();
    json.remove('cloudApiKey'); // Don't store in Hive
    await _box.put(_settingsKey, json);
  }

  // Sessions
  List<Session> loadSessions() {
    final json = _box.get(_sessionsKey);
    if (json == null) return [];
    
    try {
      final list = json as List;
      return list.map((e) => Session.fromJson(e as Map<String, dynamic>)).toList();
    } catch (e) {
      return [];
    }
  }

  Future<void> saveSessions(List<Session> sessions) async {
    final json = sessions.map((s) => s.toJson()).toList();
    await _box.put(_sessionsKey, json);
  }

  Future<void> deleteSession(String sessionId) async {
    final sessions = loadSessions();
    sessions.removeWhere((s) => s.id == sessionId);
    await saveSessions(sessions);
  }

  // Moods
  List<MoodEntry> loadMoods() {
    final json = _box.get(_moodsKey);
    if (json == null) return [];
    
    try {
      final list = json as List;
      return list.map((e) => MoodEntry.fromJson(e as Map<String, dynamic>)).toList();
    } catch (e) {
      return [];
    }
  }

  Future<void> saveMood(MoodEntry entry) async {
    final moods = loadMoods();
    final index = moods.indexWhere((m) => m.date == entry.date);
    if (index >= 0) {
      moods[index] = entry;
    } else {
      moods.add(entry);
    }
    await _box.put(_moodsKey, moods.map((m) => m.toJson()).toList());
  }

  Future<void> deleteMood(String date) async {
    final moods = loadMoods();
    moods.removeWhere((m) => m.date == date);
    await _box.put(_moodsKey, moods.map((m) => m.toJson()).toList());
  }

  // Health Samples
  List<HealthSample> loadHealthSamples() {
    final json = _box.get(_healthKey);
    if (json == null) return [];
    
    try {
      final list = json as List;
      return list.map((e) => HealthSample.fromJson(e as Map<String, dynamic>)).toList();
    } catch (e) {
      return [];
    }
  }

  Future<void> saveHealthSamples(List<HealthSample> samples) async {
    final existing = loadHealthSamples();
    final merged = [...existing, ...samples];
    await _box.put(_healthKey, merged.map((s) => s.toJson()).toList());
  }

  Future<void> saveHealthSample(HealthSample sample) async {
    final samples = loadHealthSamples();
    samples.add(sample);
    await _box.put(_healthKey, samples.map((s) => s.toJson()).toList());
  }

  // Backup/Restore
  Future<String> exportAll() async {
    final data = {
      'settings': (await loadSettings()).toJson(),
      'sessions': loadSessions().map((s) => s.toJson()).toList(),
      'moods': loadMoods().map((m) => m.toJson()).toList(),
      'health': loadHealthSamples().map((h) => h.toJson()).toList(),
    };
    return jsonEncode(data);
  }

  Future<void> importAll(String json) async {
    final data = jsonDecode(json) as Map<String, dynamic>;
    
    if (data['settings'] != null) {
      await saveSettings(AppSettings.fromJson(data['settings'] as Map<String, dynamic>));
    }
    
    if (data['sessions'] != null) {
      final sessions = (data['sessions'] as List)
          .map((e) => Session.fromJson(e as Map<String, dynamic>))
          .toList();
      await saveSessions(sessions);
    }
    
    if (data['moods'] != null) {
      final moods = (data['moods'] as List)
          .map((e) => MoodEntry.fromJson(e as Map<String, dynamic>))
          .toList();
      await _box.put(_moodsKey, moods.map((m) => m.toJson()).toList());
    }
    
    if (data['health'] != null) {
      final health = (data['health'] as List)
          .map((e) => HealthSample.fromJson(e as Map<String, dynamic>))
          .toList();
      await _box.put(_healthKey, health.map((h) => h.toJson()).toList());
    }
  }

  Future<void> clearAll() async {
    await _box.deleteAll();
    await _secureStorage.deleteApiKey('cloud_api_key');
  }

  int get approximateSize {
    return _box.length * 1024; // Rough estimate
  }
}

class HealthSample {
  final String type;
  final double value;
  final String unit;
  final DateTime timestamp;
  final String source;

  HealthSample({
    required this.type,
    required this.value,
    required this.unit,
    DateTime? timestamp,
    this.source = 'Unknown',
  }) : timestamp = timestamp ?? DateTime.now();

  HealthSample copyWith({
    String? type,
    double? value,
    String? unit,
    DateTime? timestamp,
    String? source,
  }) {
    return HealthSample(
      type: type ?? this.type,
      value: value ?? this.value,
      unit: unit ?? this.unit,
      timestamp: timestamp ?? this.timestamp,
      source: source ?? this.source,
    );
  }

  Map<String, dynamic> toJson() {
    return {
      'type': type,
      'value': value,
      'unit': unit,
      'timestamp': timestamp.toIso8601String(),
      'source': source,
    };
  }

  factory HealthSample.fromJson(Map<String, dynamic> json) {
    return HealthSample(
      type: json['type'] as String,
      value: (json['value'] as num).toDouble(),
      unit: json['unit'] as String,
      timestamp: DateTime.parse(json['timestamp'] as String),
      source: json['source'] as String? ?? 'Unknown',
    );
  }
}

class Message {
  final String id;
  final String role; // 'user' | 'assistant'
  final String content;
  final String? imageBase64;
  final DateTime timestamp;

  Message({
    required this.id,
    required this.role,
    required this.content,
    this.imageBase64,
    DateTime? timestamp,
  }) : timestamp = timestamp ?? DateTime.now();

  Message copyWith({
    String? id,
    String? role,
    String? content,
    String? imageBase64,
    DateTime? timestamp,
  }) {
    return Message(
      id: id ?? this.id,
      role: role ?? this.role,
      content: content ?? this.content,
      imageBase64: imageBase64 ?? this.imageBase64,
      timestamp: timestamp ?? this.timestamp,
    );
  }

  Map<String, dynamic> toJson() {
    return {
      'id': id,
      'role': role,
      'content': content,
      'imageBase64': imageBase64,
      'timestamp': timestamp.toIso8601String(),
    };
  }

  factory Message.fromJson(Map<String, dynamic> json) {
    return Message(
      id: json['id'] as String,
      role: json['role'] as String,
      content: json['content'] as String,
      imageBase64: json['imageBase64'] as String?,
      timestamp: DateTime.parse(json['timestamp'] as String),
    );
  }

  Map<String, dynamic> toApiMessage() {
    final msg = <String, dynamic>{
      'role': role,
      'content': content,
    };
    if (imageBase64 != null && imageBase64!.isNotEmpty) {
      msg['image_url'] = {'url': "data:image/jpeg;base64,$imageBase64"};
    }
    return msg;
  }
}

class MoodEntry {
  final String date; // 'YYYY-MM-DD'
  final int mood; // 1-5

  static const Map<int, String> emojiMap = {
    1: '😔',
    2: '😕',
    3: '😐',
    4: '🙂',
    5: '😊',
  };

  static const Map<int, String> labelMap = {
    1: 'Poor',
    2: 'Fair',
    3: 'Okay',
    4: 'Good',
    5: 'Great',
  };

  static const Map<int, int> colorMap = {
    1: 0xFFEF4444, // Red
    2: 0xFFF59E0B, // Amber
    3: 0xFF6B7280, // Gray
    4: 0xFF10B981, // Green
    5: 0xFF3B82F6, // Blue
  };

  MoodEntry({required this.date, required this.mood});

  bool get isValid => mood >= 1 && mood <= 5;

  String? get emoji => emojiMap[mood];
  String? get label => labelMap[mood];
  int? get color => colorMap[mood];

  MoodEntry copyWith({String? date, int? mood}) {
    return MoodEntry(date: date ?? this.date, mood: mood ?? this.mood);
  }

  Map<String, dynamic> toJson() {
    return {'date': date, 'mood': mood};
  }

  factory MoodEntry.fromJson(Map<String, dynamic> json) {
    return MoodEntry(
      date: json['date'] as String,
      mood: json['mood'] as int,
    );
  }
}

class Session {
  final String id;
  final String title;
  final DateTime date;
  final List<Message> messages;

  Session({
    required this.id,
    required this.title,
    required this.date,
    required this.messages,
  });

  static List<Session> enforceLimit(List<Session> sessions, {int max = 50}) {
    if (sessions.length <= max) return sessions;
    return sessions.take(max).toList();
  }

  Session copyWith({
    String? id,
    String? title,
    DateTime? date,
    List<Message>? messages,
  }) {
    return Session(
      id: id ?? this.id,
      title: title ?? this.title,
      date: date ?? this.date,
      messages: messages ?? this.messages,
    );
  }

  Map<String, dynamic> toJson() {
    return {
      'id': id,
      'title': title,
      'date': date.toIso8601String(),
      'messages': messages.map((m) => m.toJson()).toList(),
    };
  }

  factory Session.fromJson(Map<String, dynamic> json) {
    return Session(
      id: json['id'] as String,
      title: json['title'] as String,
      date: DateTime.parse(json['date'] as String),
      messages: (json['messages'] as List)
          .map((m) => Message.fromJson(m as Map<String, dynamic>))
          .toList(),
    );
  }
}

class _AppLocalizationsDelegate extends LocalizationsDelegate<AppLocalizations> {
  const _AppLocalizationsDelegate();

  @override
  Future<AppLocalizations> load(Locale locale) async {
    switch (locale.languageCode) {
      case 'de':
        return AppLocalizationsDe();
      case 'es':
        return AppLocalizationsEs();
      case 'fr':
        return AppLocalizationsFr();
      case 'hi':
        return AppLocalizationsHi();
      case 'ru':
        return AppLocalizationsRu();
      case 'zh':
        return AppLocalizationsZh();
      default:
        return AppLocalizationsEn();
    }
  }

  @override
  bool isSupported(Locale locale) {
    return AppLocalizations.supportedLocales
        .map((l) => l.languageCode)
        .contains(locale.languageCode);
  }

  @override
  bool shouldReload(_AppLocalizationsDelegate old) => false;
}

import 'package:flutter_riverpod/flutter_riverpod.dart';
import '../../core/models/mood_entry.dart';
import '../../core/storage_service.dart';

final calendarProvider = StateNotifierProvider<CalendarNotifier, CalendarState>((ref) {
  return CalendarNotifier(ref.read(storageServiceProvider));
});

class CalendarState {
  final DateTime displayedMonth;
  final Map<String, MoodEntry> moods;

  CalendarState({
    DateTime? displayedMonth,
    Map<String, MoodEntry>? moods,
  })  : displayedMonth = displayedMonth ?? DateTime.now(),
        moods = moods ?? {};

  double get monthlyAverage {
    if (moods.isEmpty) return 0;
    final values = moods.values.map((m) => m.mood.toDouble());
    return values.reduce((a, b) => a + b) / values.length;
  }

  String? get averageEmoji {
    final avg = monthlyAverage;
    if (avg < 2) return MoodEntry.emojiMap[1];
    if (avg < 3) return MoodEntry.emojiMap[2];
    if (avg < 4) return MoodEntry.emojiMap[3];
    if (avg < 5) return MoodEntry.emojiMap[4];
    return MoodEntry.emojiMap[5];
  }

  CalendarState copyWith({
    DateTime? displayedMonth,
    Map<String, MoodEntry>? moods,
  }) {
    return CalendarState(
      displayedMonth: displayedMonth ?? this.displayedMonth,
      moods: moods ?? this.moods,
    );
  }
}

class CalendarNotifier extends StateNotifier<CalendarState> {
  final StorageService _storage;

  CalendarNotifier(this._storage) : super(CalendarState()) {
    _loadMoods();
  }

  Future<void> _loadMoods() async {
    final moods = _storage.loadMoods();
    final moodMap = {for (final m in moods) m.date: m};
    state = state.copyWith(moods: moodMap);
  }

  void setMonth(DateTime month) {
    state = state.copyWith(displayedMonth: month);
  }

  Future<void> setMood(DateTime date, int mood) async {
    final dateStr = '${date.year}-${date.month.toString().padLeft(2, '0')}-${date.day.toString().padLeft(2, '0')}';
    final entry = MoodEntry(date: dateStr, mood: mood);
    await _storage.saveMood(entry);
    
    final newMoods = Map<String, MoodEntry>.from(state.moods);
    newMoods[dateStr] = entry;
    state = state.copyWith(moods: newMoods);
  }

  Future<void> clearMood(DateTime date) async {
    final dateStr = '${date.year}-${date.month.toString().padLeft(2, '0')}-${date.day.toString().padLeft(2, '0')}';
    await _storage.deleteMood(dateStr);
    
    final newMoods = Map<String, MoodEntry>.from(state.moods);
    newMoods.remove(dateStr);
    state = state.copyWith(moods: newMoods);
  }
}

import 'package:flutter_riverpod/flutter_riverpod.dart';
import '../../core/api_client.dart';
import '../../core/models/message.dart';
import '../../core/models/session.dart';
import '../../core/storage_service.dart';

final chatProvider = StateNotifierProvider<ChatNotifier, ChatState>((ref) {
  return ChatNotifier(ref.read(apiClientProvider), ref.read(storageServiceProvider));
});

class ChatState {
  final List<Session> sessions;
  final String? currentSessionId;
  final bool isLoading;
  final String? error;

  ChatState({
    List<Session>? sessions,
    this.currentSessionId,
    this.isLoading = false,
    this.error,
  }) : sessions = sessions ?? [];

  Session? get currentSession {
    if (currentSessionId == null) return null;
    try {
      return sessions.firstWhere((s) => s.id == currentSessionId);
    } catch (e) {
      return null;
    }
  }

  ChatState copyWith({
    List<Session>? sessions,
    String? currentSessionId,
    bool? isLoading,
    String? error,
  }) {
    return ChatState(
      sessions: sessions ?? this.sessions,
      currentSessionId: currentSessionId ?? this.currentSessionId,
      isLoading: isLoading ?? this.isLoading,
      error: error,
    );
  }
}

class ChatNotifier extends StateNotifier<ChatState> {
  final ApiClient _api;
  final StorageService _storage;

  ChatNotifier(this._api, this._storage) : super(ChatState()) {
    _loadSessions();
  }

  Future<void> _loadSessions() async {
    final sessions = _storage.loadSessions();
    state = state.copyWith(sessions: sessions);
  }

  Future<void> createSession() async {
    final newSession = Session(
      id: DateTime.now().millisecondsSinceEpoch.toString(),
      title: 'New Chat',
      date: DateTime.now(),
      messages: [],
    );
    
    final sessions = [...state.sessions, newSession];
    await _storage.saveSessions(sessions);
    state = state.copyWith(sessions: sessions, currentSessionId: newSession.id);
  }

  Future<void> selectSession(String sessionId) async {
    state = state.copyWith(currentSessionId: sessionId);
  }

  Future<void> deleteSession(String sessionId) async {
    await _storage.deleteSession(sessionId);
    final sessions = _storage.loadSessions();
    final newCurrentId = state.currentSessionId == sessionId ? null : state.currentSessionId;
    state = state.copyWith(sessions: sessions, currentSessionId: newCurrentId);
  }

  Future<void> sendMessage(String content, {String? imageBase64}) async {
    final session = state.currentSession;
    if (session == null) {
      await createSession();
      return sendMessage(content, imageBase64: imageBase64);
    }

    final userMessage = Message(
      id: DateTime.now().millisecondsSinceEpoch.toString(),
      role: 'user',
      content: content,
      imageBase64: imageBase64,
    );

    final updatedMessages = [...session.messages, userMessage];
    final updatedSession = session.copyWith(messages: updatedMessages);
    
    final updatedSessions = state.sessions.map((s) => s.id == session.id ? updatedSession : s).toList();
    await _storage.saveSessions(updatedSessions);
    
    state = state.copyWith(sessions: updatedSessions, isLoading: true);

    try {
      final settings = await _storage.loadSettings();
      final apiMessages = updatedMessages.map((m) => m.toApiMessage()).toList();
      
      final response = await _api.completeChat(
        messages: apiMessages,
        serverUrl: settings.serverUrl,
        useCloud: settings.useCloudApi,
        cloudApiKey: settings.cloudApiKey,
        cloudModelId: settings.cloudModelId,
        enableResearch: settings.enableResearch,
      );

      final assistantMessage = Message(
        id: DateTime.now().millisecondsSinceEpoch.toString(),
        role: 'assistant',
        content: response,
      );

      final finalMessages = [...updatedMessages, assistantMessage];
      final finalSession = updatedSession.copyWith(messages: finalMessages);
      
      final finalSessions = updatedSessions.map((s) => s.id == session.id ? finalSession : s).toList();
      await _storage.saveSessions(finalSessions);
      
      state = state.copyWith(sessions: finalSessions, isLoading: false);
    } catch (e) {
      state = state.copyWith(isLoading: false, error: e.toString());
    }
  }

  Future<void> clearChat() async {
    final session = state.currentSession;
    if (session == null) return;

    final clearedSession = session.copyWith(messages: []);
    final updatedSessions = state.sessions.map((s) => s.id == session.id ? clearedSession : s).toList();
    await _storage.saveSessions(updatedSessions);
    state = state.copyWith(sessions: updatedSessions);
  }
}

import 'package:flutter/material.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';
import '../../core/i18n/app_localizations.dart';
import '../../core/models/app_settings.dart';
import '../settings/settings_provider.dart';

class DisclaimerScreen extends ConsumerWidget {
  const DisclaimerScreen({super.key});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final l10n = AppLocalizations.of(context)!;

    return Scaffold(
      body: SafeArea(
        child: Padding(
          padding: const EdgeInsets.all(24),
          child: Column(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
              const Icon(Icons.warning_amber_rounded, size: 80, color: Colors.orange),
              const SizedBox(height: 24),
              Text(
                l10n.disclaimerTitle,
                style: const TextStyle(fontSize: 24, fontWeight: FontWeight.bold),
                textAlign: TextAlign.center,
              ),
              const SizedBox(height: 16),
              Expanded(
                child: SingleChildScrollView(
                  child: Text(
                    l10n.disclaimerBody,
                    style: const TextStyle(fontSize: 14),
                  ),
                ),
              ),
              const SizedBox(height: 24),
              ElevatedButton(
                onPressed: () => _acceptDisclaimer(ref),
                style: ElevatedButton.styleFrom(
                  minimumSize: const Size(double.infinity, 48),
                ),
                child: Text(l10n.disclaimerAccept),
              ),
            ],
          ),
        ),
      ),
    );
  }

  void _acceptDisclaimer(WidgetRef ref) {
    final currentSettings = ref.read(settingsProvider);
    ref.read(settingsProvider.notifier).updateSettings(
      currentSettings.copyWith(disclaimerAccepted: true),
    );
  }
}
